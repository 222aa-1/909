## 🛠️ 分析工具（Python脚本）

```python
# xiaohongshu_analyzer.py
import json
from collections import Counter
import re
from datetime import datetime, timedelta

class XiaohongshuAnalyzer:
    def __init__(self):
        self.keyword_categories = {
            '情感共鸣': ['破防', '泪目', '谁懂', '共情', 'emo', '绝了', '宝藏', '惊艳'],
            '实用价值': ['保姆级', '手把手', '小白', '零基础', '避雷', '踩坑', '省钱', '平替'],
            '视觉吸引': ['绝美', '神仙', '氛围感', '高级', 'ins风', '治愈', '复古'],
            '话题争议': ['大胆开麦', '真实评价', '内行人', '揭秘', '争议', '吵起来'],
            '生活分享': ['日常', '碎片', '治愈', '自律', '打卡', '经验', '复盘']
        }
    
    def analyze_notes(self, notes_data):
        """分析小红书笔记数据"""
        results = {
            'total_notes': len(notes_data),
            'time_range': '近15天',
            'analysis_date': datetime.now().strftime('%Y-%m-%d'),
            'keyword_stats': {},
            'category_stats': {},
            'top_titles': [],
            'insights': []
        }
        
        # 提取所有文案内容
        all_content = ' '.join([note.get('content', '') for note in notes_data])
        
        # 关键词分析
        keyword_counts = Counter()
        for category, keywords in self.keyword_categories.items():
            category_count = 0
            for keyword in keywords:
                count = all_content.count(keyword)
                keyword_counts[keyword] = count
                category_count += count
            
            results['category_stats'][category] = category_count
        
        # 获取热门关键词
        top_keywords = keyword_counts.most_common(20)
        results['keyword_stats'] = dict(top_keywords)
        
        # 分析标题特点
        titles = [note.get('title', '') for note in notes_data]
        results['top_titles'] = sorted(titles, key=len, reverse=True)[:10]
        
        # 生成洞察
        results['insights'] = self.generate_insights(results)
        
        return results
    
    def generate_insights(self, analysis_results):
        """生成分析洞察"""
        insights = []
        
        # 基于关键词分析
        top_categories = sorted(
            analysis_results['category_stats'].items(),
            key=lambda x: x[1],
            reverse=True
        )[:3]
        
        if top_categories:
            insights.append(f"🔥 热门内容类型: {', '.join([cat for cat, _ in top_categories])}")
        
        # 标题分析
        if analysis_results['top_titles']:
            avg_title_length = sum(len(title) for title in analysis_results['top_titles']) / len(analysis_results['top_titles'])
            insights.append(f"📝 平均标题长度: {avg_title_length:.1f}字")
        
        # 关键词建议
        top_keywords = list(analysis_results['keyword_stats'].keys())[:5]
        insights.append(f"🔑 热门关键词: {', '.join(top_keywords)}")
        
        return insights
    
    def create_report(self, analysis_results):
        """生成分析报告"""
        report = []
        report.append("# 小红书爆款文案分析报告")
        report.append(f"## 分析时间: {analysis_results['analysis_date']}")
        report.append(f"## 分析范围: {analysis_results['time_range']}")
        report.append(f"## 分析样本: {analysis_results['total_notes']}篇笔记")
        report.append("")
        
        report.append("## 📊 关键词统计")
        report.append("| 关键词 | 出现次数 | 类别 |")
        report.append("|--------|----------|------|")
        
        for keyword, count in analysis_results['keyword_stats'].items():
            # 查找关键词所属类别
            category = '其他'
            for cat, keywords in self.keyword_categories.items():
                if keyword in keywords:
                    category = cat
                    break
            
            report.append(f"| {keyword} | {count} | {category} |")
        
        report.append("")
        
        report.append("## 🏆 内容类型排名")
        report.append("| 内容类型 | 出现次数 | 占比 |")
        report.append("|----------|----------|------|")
        
        total_keywords = sum(analysis_results['category_stats'].values())
        for category, count in sorted(
            analysis_results['category_stats'].items(),
            key=lambda x: x[1],
            reverse=True
        ):
            percentage = (count / total_keywords * 100) if total_keywords > 0 else 0
            report.append(f"| {category} | {count} | {percentage:.1f}% |")
        
        report.append("")
        
        report.append("## 📝 热门标题示例")
        for i, title in enumerate(analysis_results['top_titles'][:5], 1):
            report.append(f"{i}. {title}")
        
        report.append("")
        
        report.append("## 💡 分析洞察")
        for insight in analysis_results['insights']:
            report.append(f"- {insight}")
        
        report.append("")
        
        report.append("## 🚀 内容创作建议")
        report.append("### 1. 标题优化")
        report.append("- 使用数字吸引注意力（如'3个技巧'、'5分钟学会'）")
        report.append("- 加入情感词汇（如'破防了'、'绝了'）")
        report.append("- 设置悬念（如'没想到...'、'原来是这样'）")
        
        report.append("### 2. 内容策略")
        report.append("- 提供实用价值（教程、避坑指南）")
        report.append("- 引发情感共鸣（分享真实经历）")
        report.append("- 创造视觉冲击（高质量图片/视频）")
        
        report.append("### 3. 互动提升")
        report.append("- 结尾提问引导评论")
        report.append("- 使用投票增加互动")
        report.append("- 定期举办抽奖活动")
        
        return '\n'.join(report)
```
